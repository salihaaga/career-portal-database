# career-portal-database
Bu projede, online iş veya staj bulma imkanı sağlanıyor. Tablolar arası ilişki kurulmuştur. Bu projede sql dili kullanılmıştır.

REM   Script: VTYS TRIGGERS
REM   TRIGGERS

CREATE TABLE ADAYLAR ( 
    ADAY_ID INT PRIMARY KEY, 
    AD VARCHAR(50), 
    SOYAD VARCHAR(50), 
    TELEFON VARCHAR(20), 
    EMAIL VARCHAR(100), 
    DOGUM_TARIHI DATE 
);

CREATE TABLE YETENEKLER ( 
    YETENEK_ID INT PRIMARY KEY, 
    YETENEK_ADI VARCHAR(100) 
);

CREATE TABLE FIRMALAR ( 
    FIRMA_ID INT PRIMARY KEY, 
    FIRMA_ADI VARCHAR(100), 
    EMAIL VARCHAR(100), 
    TELEFON VARCHAR(20) 
);

CREATE TABLE ILAN ( 
    ILAN_ID INT PRIMARY KEY, 
    BASLIK VARCHAR(100), 
    FIRMA_ID_I INT, 
    POZISYON VARCHAR(100), 
    KONUM VARCHAR(100), 
    YAYINLANMA_TARIHI DATE, 
    SON_BASVURU_TARIHI DATE, 
    FOREIGN KEY (FIRMA_ID_I) REFERENCES FIRMALAR(FIRMA_ID) 
);

CREATE TABLE EGITIM_BILGISI ( 
    EB_ID INT PRIMARY KEY, 
    EGITIM_DURUMU VARCHAR(100), 
    BOLUM VARCHAR(100), 
    ADAY_ID_EB INT UNIQUE, 
    FOREIGN KEY (ADAY_ID_EB) REFERENCES ADAYLAR(ADAY_ID) 
);

CREATE TABLE DENEYIMLER ( 
    DENEYIM_ID INT PRIMARY KEY, 
    DENEYIM_SURE VARCHAR(50), 
    ACIKLAMA VARCHAR(100), 
    ADAY_ID_D INT, 
    FOREIGN KEY (ADAY_ID_D) REFERENCES ADAYLAR(ADAY_ID) 
);

CREATE TABLE ADAY_YETENEK ( 
    ADAY_ID_AY INT, 
    YETENEK_ID_AY INT, 
    YETENEK_DUZEYI VARCHAR(20), 
    PRIMARY KEY (ADAY_ID_AY, YETENEK_ID_AY), 
    FOREIGN KEY (ADAY_ID_AY) REFERENCES ADAYLAR(ADAY_ID), 
    FOREIGN KEY (YETENEK_ID_AY) REFERENCES YETENEKLER(YETENEK_ID) 
);

CREATE TABLE ILAN_YETENEK ( 
    ILAN_ID_IY INT, 
    YETENEK_ID_IY INT, 
    GEREKEN_DUZEY VARCHAR(20), 
    PRIMARY KEY (ILAN_ID_IY, YETENEK_ID_IY), 
    FOREIGN KEY (ILAN_ID_IY) REFERENCES ILAN(ILAN_ID), 
    FOREIGN KEY (YETENEK_ID_IY) REFERENCES YETENEKLER(YETENEK_ID) 
);

CREATE TABLE BASVURULAR ( 
    BASVURU_ID INT PRIMARY KEY, 
    BASVURU_TARIHI DATE, 
    DURUMU VARCHAR(20), 
    ADAY_ID_B INT UNIQUE, 
    ILAN_ID INT, 
    FOREIGN KEY (ADAY_ID_B) REFERENCES ADAYLAR(ADAY_ID), 
    FOREIGN KEY (ILAN_ID) REFERENCES ILAN(ILAN_ID) 
);

CREATE OR REPLACE TRIGGER trg_ilan_tarih_kontrol 
  BEFORE INSERT OR UPDATE ON ILAN 
  FOR EACH ROW 
BEGIN 
  IF :NEW.SON_BASVURU_TARIHI < :NEW.YAYINLANMA_TARIHI THEN 
    RAISE_APPLICATION_ERROR(-20001, 
      'Hata: Son başvuru tarihi, yayınlanma tarihinden önce olamaz.'); 
END IF; 
END;

CREATE TABLE ADAYLAR_LOG ( 
  LOG_ID         NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY, 
  ADAY_ID        NUMBER, 
  ISLEM_TURU     VARCHAR2(10),  -- 'INSERT', 'UPDATE', 'DELETE' 
  ISLEM_ZAMANI   TIMESTAMP, 
  AD            VARCHAR2(100), 
  SOYAD         VARCHAR2(100), 
  EMAIL         VARCHAR2(100), 
  TELEFON       VARCHAR2 (20), 
  DOGUM_TARIHI  DATE 
);

CREATE OR REPLACE TRIGGER adaylar_log 
  AFTER INSERT OR UPDATE OR DELETE ON ADAYLAR 
  FOR EACH ROW 
BEGIN 
  IF INSERTING THEN 
    INSERT INTO ADAYLAR_LOG (ADAY_ID, ISLEM_TURU, ISLEM_ZAMANI, AD, SOYAD, EMAIL, TELEFON, DOGUM_TARIHI) 
    VALUES (:NEW.ADAY_ID, 'INSERT', SYSTIMESTAMP, :NEW.AD, :NEW.SOYAD, :NEW.EMAIL, :NEW.TELEFON, :NEW.DOGUM_TARIHI); 
 
  ELSIF UPDATING THEN 
    INSERT INTO ADAYLAR_LOG (ADAY_ID, ISLEM_TURU, ISLEM_ZAMANI, AD, SOYAD, EMAIL, TELEFON, DOGUM_TARIHI) 
    VALUES (:NEW.ADAY_ID, 'UPDATE', SYSTIMESTAMP, :NEW.AD, :NEW.SOYAD, :NEW.EMAIL, :NEW.TELEFON, :NEW.DOGUM_TARIHI); 
 
  ELSIF DELETING THEN 
    INSERT INTO ADAYLAR_LOG (ADAY_ID, ISLEM_TURU, ISLEM_ZAMANI, AD, SOYAD, EMAIL, TELEFON, DOGUM_TARIHI) 
    VALUES (:NEW.ADAY_ID, 'DELETE', SYSTIMESTAMP, :NEW.AD, :NEW.SOYAD, :NEW.EMAIL, :NEW.TELEFON, :NEW.DOGUM_TARIHI); 
  END IF; 
END; 
/

CREATE OR REPLACE TRIGGER trg_firma_sil 
BEFORE DELETE ON FIRMALAR 
FOR EACH ROW 
BEGIN 
  -- Önce ilana bağlı ilan_yetenek kayıtlarını sil 
  DELETE FROM ILAN_YETENEK  
  WHERE ILAN_ID_IY IN ( 
    SELECT ILAN_ID FROM ILAN WHERE FIRMA_ID_I = :OLD.FIRMA_ID 
  ); 
 
  -- Sonra ilana bağlı başvuru kayıtlarını sil 
  DELETE FROM BASVURULAR  
  WHERE ILAN_ID IN ( 
    SELECT ILAN_ID FROM ILAN WHERE FIRMA_ID_I = :OLD.FIRMA_ID 
  ); 
 
  -- En son ilanları sil 
  DELETE FROM ILAN  
  WHERE FIRMA_ID_I = :OLD.FIRMA_ID; 
END; 
/

CREATE OR REPLACE TRIGGER trg_basvuru_tarihi_kontrol 
  BEFORE INSERT ON BASVURULAR 
  FOR EACH ROW 
DECLARE 
  v_son_tarih DATE; 
BEGIN 
  SELECT SON_BASVURU_TARIHI INTO v_son_tarih 
  FROM ILAN 
  WHERE ILAN_ID = :NEW.ILAN_ID; 
 
  IF v_son_tarih < TRUNC(SYSDATE) THEN 
    RAISE_APPLICATION_ERROR(-20002, 
      'Hata: Bu ilanın son başvuru tarihi geçmiştir. Başvuru yapılamaz.'); 
END IF; 
END;
